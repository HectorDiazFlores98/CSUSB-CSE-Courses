<script language="javascript" type="text/javascript">

function findMinMax (array){
let minVar=array[0];
let maxVar=array[0];

for(let i = 0; i <= array.length; i++)
{
if(minVar<=array[i])
{
minVar=array[i];
}
else if(maxVar>=array[i])
{
maxVar=array[i];
}
}

console.log('Min value= '+minVar);
console.log('Max value= '+maxVar);
}

console.log(findMinMax([1,4,-5,-4,30,2]));


</script>
